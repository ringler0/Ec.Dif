CREATE TABLE datos_uma (
	id_uma SERIAL,
	time varchar(30),
	uma int8,
	lat float8,
	lon float8,
	mp01 int8,
	mp25 int8,
	mp10 int8,
	d03 int8,
	d05 int8,
	d01 int8,
	d25 int8,
	d50 int8,
	d10 int8,
	vel float8,
	PRIMARY KEY (id_uma)
);
